DDL Code  CREATE TABLE appeal(
     appeal_id                                                 VARCHAR(50)        NOT NULL,
     denial_id                                                 VARCHAR(50),
     claim_id                                                  VARCHAR(50),
     appeal_level                                              NUMBER(1, 0),
     "-- 1st, 2nd, 3rd level
     appeal_date"                  DATE,
     appeal_reason                                             TEXT(10),
     appeal_status                                             VARCHAR(20),
     "-- Pending, Approved, Denied
     appeal_decision_date"   DATE,
     appeal_outcome_amount                                     DECIMAL(12, 2),
     created_date                                              CHAR(10)           DEFAULT CURRENT_TIMESTAMP(),
     updated_date                                              CHAR(10)           DEFAULT CURRENT_TIMESTAMP(),
     CONSTRAINT PK15 PRIMARY KEY (appeal_id),
     CONSTRAINT Refclaim28 FOREIGN KEY (claim_id)
     REFERENCES claim(claim_id),
     CONSTRAINT Refdenial27 FOREIGN KEY (denial_id)
     REFERENCES denial(denial_id)
)
;
